  • Explain how "fillList2" is more efficient than "fillList". 
    var createListItem = function (data, index) {
    var li = document.createElement("li"),
        node = document.createTextNode(index + ". " + data.text);
    li.appendChild(node);
    return li;
};

var fillList = function (list) {
    var ul = document.getElementById("myUl"),
        i = 0,
        len = list.length;
    for (; i < len; i++) {
        ul.appendChild(createListItem(list[i], i));
    }
};

var fillList2 = function (list) {
    var ul = document.getElementById("myUl"),
        fragment = document.createDocumentFragment(),
        i = 0,
        len = list.length;
    for (; i < len; i++) {
        fragment.appendChild(createListItem(list[i], i));
    }
    ul.appendChild(fragment);
};
